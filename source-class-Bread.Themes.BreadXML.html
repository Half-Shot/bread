 53:     public function convertObjtoElement($obj,\DOMNode $root,\DOMDocument $doc,$id = NULL)
 54: {
 55:         $en = $id;
 56:         if($en == NULL)
 57:            $en = self::ELEMENT_VARIABLE;
 58:         $objRoot = $doc->createElement($en);
 59:         if(is_object($obj) || is_array($obj))
 60:         {
 61:             if($id !== NULL)
 62:                 $objRoot->setAttribute ("id", $id);
 63:             foreach($obj as $objtype => $value){
 64:                 if(is_numeric($objtype))
 65:                     $objtype = NULL;
 66:                 $objRoot->appendChild($this->convertObjtoElement($value,$objRoot,$doc,$objtype));
 67:             }
 68:             return $objRoot;
 69:         }
 70:         else
 71:         {
 72:             if(is_string($obj)){
 73:                 $objRoot->appendChild ($doc->createTextNode ($obj));
 74:             }
 75:             elseif(is_bool($obj))
 76:             {
 77:                 $objRoot->setAttribute("type", "xs:boolean");
 78:                 if($obj){
 79:                     $objRoot->nodeValue = 1;
 80:                 }
 81:                 else{
 82:                     $objRoot->nodeValue = 0;
 83:                 } 
 84:             }
 85:             else
 86:             {
 87:                 $objRoot->nodeValue = (string)$obj;
 88:             }
 89:             if($id !== NULL)
 90:                 $objRoot->setAttribute ("id", $id);
 91:             return $objRoot;
 92:         }
 93:     }

 95:     public function GetHTMLOfElement($ElementId,$Variables = array(),$ElementType = "telement")
 96:     {
 97:         $xmldoc = new \DOMDocument('1.0', 'utf-8');
 98:         $root = $xmldoc->createElement("root");
 99:         $xmldoc->appendChild($root);

101:         $elementStuff->setAttribute("id", $ElementId);
102:         $elementStuff->appendChild($this->convertObjtoElement($Variables,$elementStuff,$xmldoc));
