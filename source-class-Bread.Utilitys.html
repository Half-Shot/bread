 52:         public static function DigestURL($url)
 53:         {
 54:             
 55:             $parts = \explode("?",$url);
 56:             $baseURL = $parts[0];
 57:             if(count($parts) > 1){
 58:                 $parts = \explode("&",$parts[1]);
 59:             }
 60:             $returnedArray = array();
 61:             $returnedArray["BASEURL"] = $baseURL;
 62:             foreach($parts as $part)
 63:             {
 64:                $pair = \explode("=",$part);
 65:                if(count($pair) > 1)
 66:                 $returnedArray[$pair[0]] = $pair[1];

 80:         {
 81:             if(!$baseurl)

 85:             if(array_count_values($params) < 1)
 86:                 return $url;

 88:             $url .= "?" . $key . "=" .$params[$key];

 91:                 return $url;

317:         static function curl_exec_follow($ch, &$maxredirect = null) {
318: 
319:           $mr = $maxredirect === null ? 5 : intval($maxredirect);
320: 
321:           if (ini_get('open_basedir') == '' && ini_get('safe_mode') == 'Off') {
322: 
323:             curl_setopt($ch, CURLOPT_FOLLOWLOCATION, $mr > 0);
324:             curl_setopt($ch, CURLOPT_MAXREDIRS, $mr);
325:             curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
326:             curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
327: 
328:           } else {
329: 
330:             curl_setopt($ch, CURLOPT_FOLLOWLOCATION, false);
331: 
332:             if ($mr > 0)
333:             {
334:               $original_url = curl_getinfo($ch, CURLINFO_EFFECTIVE_URL);
335:               $newurl = $original_url;
336: 
337:               $rch = curl_copy_handle($ch);
338: 
339:               curl_setopt($rch, CURLOPT_HEADER, true);
340:               curl_setopt($rch, CURLOPT_NOBODY, true);
341:               curl_setopt($rch, CURLOPT_FORBID_REUSE, false);
342:               do
343:               {
344:                 curl_setopt($rch, CURLOPT_URL, $newurl);
345:                 $header = curl_exec($rch);
346:                 if (curl_errno($rch)) {
347:                   $code = 0;
348:                 } else {
349:                   $code = curl_getinfo($rch, CURLINFO_HTTP_CODE);
350:                   if ($code == 301 || $code == 302) {
351:                     preg_match('/Location:(.*?)\n/', $header, $matches);
352:                     $newurl = trim(array_pop($matches));
353: 
354:                     // if no scheme is present then the new url is a
355:                     // relative path and thus needs some extra care
356:                     if(!preg_match("/^https?:/i", $newurl)){
357:                       $newurl = $original_url . $newurl;
358:                     }   
359:                   } else {
360:                     $code = 0;
361:                   }
362:                 }
363:               } while ($code && --$mr);
364: 
365:               curl_close($rch);
366: 
367:               if (!$mr)
368:               {
369:                 if ($maxredirect === null)
370:                 trigger_error('Too many redirects.', E_USER_WARNING);
371:                 else
372:                 $maxredirect = 0;
373: 
374:                 return false;
375:               }
376:               curl_setopt($ch, CURLOPT_URL, $newurl);
377:             }
378:           }
379:           return curl_exec($ch);

397:         /**
398:         * Copy a file, or recursively copy a folder and its contents
399:         * @param       string   $source    Source path
400:         * @param       string   $dest      Destination path
401:         * @param       string   $permissions New folder creation permissions
402:         * @return      bool     Returns true on success, false on failure
403:         * @author      mjolnic <http://stackoverflow.com/a/12763962>
404:         */
405:         static function xcopy($source, $dest, $permissions = 0755)
406:         {
407:            // Check for symlinks
408:            if (is_link($source)) {
409:                return symlink(readlink($source), $dest);
410:            }
411:            // Simple copy for a file
412:            if (is_file($source)) {
413:                return copy($source, $dest);
414:            }
415: 
416:            // Make destination directory
417:            if (!is_dir($dest)) {
418:                mkdir($dest, $permissions);
419:            }
420: 
421:            // Loop through the folder
422:            $dir = dir($source);
423:            while (false !== $entry = $dir->read()) {
424:                // Skip pointers
425:                if ($entry == '.' || $entry == '..') {
426:                    continue;
427:                }
428:                 
429:                // Deep copy directories
430:                self::xcopy("$source/$entry", "$dest/$entry");
431:            }
432: 
433:            // Clean up
434:            $dir->close();
435:            return true;
436:         }
