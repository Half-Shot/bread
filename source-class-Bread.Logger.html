580:         public static function DigestURL($url)
581:         {
582:             
583:             $parts = \explode("?",$url);
584:             $baseURL = $parts[0];
585:             if(count($parts) > 1){
586:                 $parts = \explode("&",$parts[1]);
587:             }
588:             $returnedArray = array();
589:             $returnedArray["BASEURL"] = $baseURL;
590:             foreach($parts as $part)
591:             {
592:                $pair = \explode("=",$part);
593:                if(count($pair) > 1)
594:                 $returnedArray[$pair[0]] = $pair[1];
595:                else

597:             }
598:             return $returnedArray;
